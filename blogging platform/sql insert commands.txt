INSERT INTO public.booklist (user_id, blog_id)
SELECT user_id, blog_id
FROM (
    -- Subquery to generate random pairs and assign a rank to each
    SELECT 
        u.user_id, 
        b.blog_id,
        row_number() OVER (PARTITION BY u.user_id ORDER BY random()) as rank
    FROM 
        public.users u
    CROSS JOIN 
        public.blogs b
) ranked_bookmarks
WHERE rank <= 6  -- This is the hard constraint: max 6 per user
ON CONFLICT (user_id, blog_id) DO NOTHING;

-- ==============================================================================
-- VERIFICATION QUERY (Run this to check the distribution)
-- ==============================================================================
SELECT 
    u.first_name || ' ' || u.last_name as user_name,
    COUNT(bl.blog_id) as total_bookmarks
FROM 
    public.users u
JOIN 
    public.booklist bl ON u.user_id = bl.user_id
GROUP BY 
    u.user_id, u.first_name, u.last_name
ORDER BY 
    total_bookmarks DESC;

-- GENERATE RANDOM DRAFTS (Max 3 per Author)
-- ==============================================================================

INSERT INTO public.blogs (title, content, author_id, status, is_premium, update_datetime)
SELECT 
    'Draft: ' || draft_topics.topic || ' (' || ranked_drafts.rank || ')',
    'This is a preliminary draft focusing on ' || draft_topics.topic || '. The author is currently researching primary sources and drafting the initial technical breakdown. Content will be expanded to 300+ words upon publication.',
    ranked_drafts.author_id,
    0, -- Status 0 = Draft
    (random() > 0.7), -- 30% chance the intended final blog is premium
    CURRENT_TIMESTAMP - (random() * 2 || ' days')::interval
FROM (
    -- Rank potential drafts to enforce the "At Max 3" rule
    SELECT 
        author_id,
        row_number() OVER (PARTITION BY author_id ORDER BY random()) as rank
    FROM (
        -- Generate potential slots (4 per author to ensure we have enough to pick from)
        SELECT generate_series(2, 16) as author_id, generate_series(1, 4)
    ) as draft_slots
) ranked_drafts
CROSS JOIN (
    -- A small pool of realistic draft titles
    SELECT unnest(ARRAY[
        'The Future of Quantum Encryption',
        'Modernism in Urban Architecture',
        'The Impact of High-Frequency Trading',
        'Ethical Implications of Neural Links',
        'The Evolution of Sub-Saharan Trade',
        'Micro-services vs Monoliths in 2026'
    ]) as topic
) draft_topics
WHERE ranked_drafts.rank <= 3 -- Hard constraint: Max 3 per author
AND random() < 0.7 -- Adds variety so some authors might have 0, 1, or 2 drafts instead of always 3
ORDER BY random()
LIMIT 30; -- Total pool of drafts to add across all authors

-- ==============================================================================
-- RESET SEQUENCE
-- ==============================================================================
SELECT setval(pg_get_serial_sequence('blogs', 'blog_id'), coalesce(max(blog_id), 1), max(blog_id) IS NOT null) FROM blogs;

INSERT INTO public.donations (user_id, blog_id, author_id, amount, donated_at)
SELECT 
    random_data.user_id,
    b.blog_id,
    b.author_id, -- This dynamically fetches the correct author for the constraint
    random_data.amount,
    random_data.donation_datetime
FROM (
    -- Step 1: Generate the random metrics
    SELECT 
        floor(random() * 30 + 1)::int AS user_id,
        floor(random() * 60 + 1)::int AS blog_id,
        round((random() * 999.99 + 0.01)::numeric, 2) AS amount,
        CURRENT_TIMESTAMP - (random() * 60 || ' days')::interval AS donation_datetime
    FROM generate_series(1, 50)
) random_data
-- Step 2: Join with the blogs table to grab the required author_id
JOIN public.blogs b ON random_data.blog_id = b.blog_id;

-- ==============================================================================
-- VERIFICATION QUERY (See who the top earners are!)
-- ==============================================================================
SELECT 
    u.first_name || ' ' || u.last_name as author_name,
    COUNT(d.donation_id) as total_donations,
    SUM(d.amount) as total_revenue
FROM 
    public.users u
JOIN 
    public.donations d ON u.user_id = d.author_id
GROUP BY 
    u.user_id, author_name
ORDER BY 
    total_revenue DESC;

-- ==============================================================================
-- RESET THE SEQUENCE 
-- ==============================================================================
SELECT setval(pg_get_serial_sequence('donations', 'donation_id'), coalesce(max(donation_id), 1), max(donation_id) IS NOT null) FROM donations;
