CREATE TABLE public.users (
    user_id integer GENERATED ALWAYS AS IDENTITY primary key,
    first_name character varying NOT NULL,
    last_name character varying,
    email_id character varying NOT NULL unique,
    password character varying NOT NULL,
    premium smallint DEFAULT '0'::smallint NOT NULL,
    avatar_id integer,
    security_question character varying(255),
    security_answer character varying(255)
);

CREATE TABLE public.categories (
    category_id integer GENERATED ALWAYS AS IDENTITY primary key,
    category_name character varying NOT NULL unique
);

CREATE TABLE public.admins (
    admin_id integer primary key,
    role character varying(50) DEFAULT 'super_admin'::character varying,
    salary integer,
    check (salary>0),
    constraint c_name26 foreign key (admin_id) references users(user_id) on delete cascade
);

CREATE TABLE public.interested_categories (
    user_id integer,
    category_id integer,
    primary key(user_id, category_id),
    constraint c_name1 foreign key (user_id) references users(user_id) on delete cascade,
    constraint c_name2 foreign key (category_id) references categories(category_id) on delete cascade
);

CREATE TABLE public.subscriptions (
    sub_id integer GENERATED ALWAYS AS IDENTITY primary key,
    user_id integer not null,
    plan_type character varying(20),
    start_date date not null,
    end_date date not null,
    amount integer,
    check (plan_type in ('monthly','yearly')),
    constraint c_name19 foreign key (user_id) references users(user_id) on delete cascade,
    check(amount>0),
    check (end_date > start_date)
);

CREATE TABLE public.authors (
    author_id integer primary key,
    publi_id integer,
    constraint c_name3 foreign key (author_id) references users(user_id) on delete cascade
);

CREATE TABLE public.publications (
    publi_id integer GENERATED ALWAYS AS IDENTITY primary key,
    publi_name character varying NOT NULL unique,
    date_of_joining date NOT NULL default CURRENT_DATE,
    owner_id integer NOT NULL,
    constraint c_name9 foreign key (owner_id) references authors(author_id) on delete cascade
);

ALTER TABLE public.authors 
    ADD constraint c_name4 foreign key (publi_id) references publications(publi_id) ON DELETE SET NULL;

CREATE TABLE public.author_followers (
    author_id integer,
    user_id integer,
    primary key (author_id,user_id),
    constraint c_name16 foreign key (user_id) references users(user_id) on delete cascade,
    constraint c_name17a foreign key (author_id) references authors(author_id) on delete cascade
);

CREATE TABLE public.publication_followers (
    publi_id integer,
    user_id integer,
    primary key (publi_id,user_id),
    constraint c_name17 foreign key (publi_id) references publications(publi_id) on delete cascade,
    constraint c_name18 foreign key (user_id) references users(user_id) on delete cascade
);

CREATE TABLE public.publication_join_requests (
    request_id integer GENERATED ALWAYS AS IDENTITY primary key,
    user_id integer,
    publi_id integer,
    status character varying(10) DEFAULT 'pending'::character varying,
    constraint c_name23 foreign key (user_id) references users(user_id) on delete cascade,
    constraint c_name24 foreign key (publi_id) references publications(publi_id) on delete cascade,
    check (status in ('pending', 'accepted', 'rejected'))
);

-- status=0 : draft, =1 : published
CREATE TABLE public.blogs (
    blog_id integer GENERATED ALWAYS AS IDENTITY primary key,
    content text NOT NULL,
    author_id integer NOT NULL,
    update_datetime timestamp without time zone NOT NULL,
    status smallint DEFAULT '0'::smallint NOT NULL,
    title text NOT NULL,
    is_premium boolean DEFAULT false,
    constraint c_name5 foreign key (author_id) references authors(author_id) on delete cascade,
    check (status in (0,1))
);

CREATE TABLE public.blog_categories (
    blog_id integer,
    category_id integer,
    primary key (blog_id, category_id),
    constraint c_name6 foreign key (blog_id) references blogs(blog_id) on delete cascade,
    constraint c_name7 foreign key (category_id) references categories(category_id) on delete cascade
);

CREATE TABLE public.blog_keywords (
    blog_id integer,
    keyword text,
    primary key(blog_id, keyword),
    constraint c_name8 foreign key (blog_id) references blogs(blog_id) on delete cascade
);

CREATE TABLE public.booklist (
    user_id integer,
    blog_id integer,
    primary key (user_id, blog_id),
    constraint c_name14 foreign key (blog_id) references blogs(blog_id) on delete cascade,
    constraint c_name15 foreign key (user_id) references users(user_id) on delete cascade
);

CREATE TABLE public.comments (
    comment_id integer GENERATED ALWAYS AS IDENTITY primary key,
    description text NOT NULL,
    blog_id integer NOT NULL,
    user_id integer NOT NULL,
    creation_datetime timestamp without time zone NOT NULL DEFAULT CURRENT_TIMESTAMP,
    constraint c_name10 foreign key (blog_id) references blogs(blog_id) on delete cascade,
    constraint c_name11 foreign key (user_id) references users(user_id) on delete cascade
);

CREATE TABLE public.donations (
    donation_id integer GENERATED ALWAYS AS IDENTITY primary key,
    user_id integer,
    author_id integer not null,
    blog_id integer,
    amount integer,
    donated_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    check (amount>0),
    constraint c_name20 foreign key (user_id) references users(user_id) on delete set null,
    constraint c_name21 foreign key (author_id) references authors(author_id) on delete cascade,
    constraint c_name22 foreign key (blog_id) references blogs(blog_id) on delete set null
);

CREATE TABLE public.likes (
    user_id integer,
    blog_id integer,
    primary key(user_id, blog_id),
    constraint c_name12 foreign key (blog_id) references blogs(blog_id) on delete cascade,
    constraint c_name13 foreign key (user_id) references users(user_id) on delete cascade
);

CREATE TABLE public.reports (
    report_id integer GENERATED ALWAYS AS IDENTITY primary key,
    user_id integer,
    blog_id integer,
    reported_at timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    type_of_report character varying NOT NULL,
    description text,
    status character varying DEFAULT 'new'::character varying,
    constraint c_name28 foreign key (user_id) references users(user_id) on delete cascade,
    constraint c_name29 foreign key (blog_id) references blogs(blog_id) on delete cascade,
    check (status in ('new','solved'))
);
